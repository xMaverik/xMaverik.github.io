def draw_cloud(t, x, y, size=1.0): """Draw a cloud made of multiple overlapping circles""" jump_to(t, x, y) t.pencolor("white") # Draw multiple overlapping circles to form a cloud positions = [(0, 0), (size * 15, size * 10), (size * 30, 0), (size * 45, size * 5), (size * 60, 0)] for pos_x, pos_y in positions: jump_to(t, x + pos_x, y + pos_y) draw_circle(t, size * 20, "white")
