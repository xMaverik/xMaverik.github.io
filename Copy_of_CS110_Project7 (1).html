print("\n--- Data Cleaning ---")

# Check for missing values
print("\nMissing values before cleaning:")
print(df.isnull().sum())

# Handle missing values
# For 'date_added', convert to datetime. Use errors='coerce' to turn invalid parsing into NaT
# and then fill NaNs. Use format='mixed' to handle variations in format.
df['date_added'] = pd.to_datetime(df['date_added'], errors='coerce', format='mixed')

# Fill missing dates with the mode AFTER conversion, which will now include NaT values
# The mode() will return a Series, so we take the first element [0]
df['date_added'].fillna(df['date_added'].mode()[0], inplace=True)


for col in ['director', 'cast', 'country', 'rating', 'duration']:
    df[col].fillna('Unknown', inplace=True)

# Check for duplicates
print(f"\nNumber of duplicate rows before cleaning: {df.duplicated().sum()}")
df.drop_duplicates(inplace=True)
print(f"Number of duplicate rows after cleaning: {df.duplicated().sum()}")

# Verify missing values after cleaning
print("\nMissing values after cleaning:")
print(df.isnull().sum())

# --- Part 2: Create Visualizations ---
print("\n--- Creating Visualizations ---")

# Visualization 1: Trends in content added over years
# Extract year from 'date_added'
df['year_added'] = df['date_added'].dt.year

# Group by year_added and type
content_by_year_type = df.groupby(['year_added', 'type']).size().unstack(fill_value=0)

plt.figure(figsize=(14, 7))
sns.lineplot(data=content_by_year_type)
plt.title('Movies and TV Shows Added to Netflix Over Years')
plt.xlabel('Year Added')
plt.ylabel('Number of Titles Added')
plt.legend(title='Content Type')
plt.grid(True, linestyle='--', alpha=0.7)
plt.show()

# Visualization 2: Top Genres
# The 'listed_in' column contains comma-separated genres. We need to split them.
genres = df['listed_in'].str.split(', ').explode()
top_genres = genres.value_counts().head(15) # Get top 15 genres

plt.figure(figsize=(12, 8))
sns.barplot(x=top_genres.index, y=top_genres.values, palette='viridis')
plt.title('Top 15 Most Popular Genres on Netflix')
plt.xlabel('Genre')
plt.ylabel('Number of Titles')
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()
